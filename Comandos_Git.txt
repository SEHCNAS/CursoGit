git ini 
	- para iniciar o git na pasta selecionada

git remote add origin + Url 
	- Defini o ambiente remoto do git

git remote get-url origin 
	- Mostra qual ambiente esta sendo usado

git config --local user.name
	- Defini o nome que sera usado

git config --local user.email 
	- Defini o email que sera usado 

git config --global alias.<Comando curto><Comando longo>
	- Defini um atalho para algum comando

git <Comando> --help ou git <Comando> -help
	- Abre a documentação do comando especificado

git add <nome do arquivo>
	-Adiciona o arquivo para ser enviado no commit

git commit -m "Mensagem obrigatoria"
	- prepara os arquivo para serem enviados

git push --set-upstream origin master
	- Comando para realizar o envio para o github 

git cat-file -t <hash>
	-Mostra oque é a hase

git cat-file -p <hash>
	-Mostra mostra o conteudo da hase

git pull
	-Pega as alterações remotas e baixa localmente
